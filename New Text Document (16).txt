title Hausdatenbank - Entity Relationship Diagram

TABLE(persons) {
    PRIMARY_KEY(id) : INT <<AUTO_INCREMENT>>
    --
    NOT_NULL(name) : VARCHAR(255)
    NOT_NULL(mobile) : VARCHAR(50)
    NOT_NULL(email) : VARCHAR(255) <<UNIQUE>>
    NOT_NULL(person_type) : ENUM
    dossier : TEXT
}

note right of persons::person_type
  ENUM Werte:
  'Gast', 'Besitzer', 
  'Familie', 'Freund', 
  'Kunde', 'Anbieter', 
  'Kursteilnehmer'
end note

TABLE(locations) {
    PRIMARY_KEY(id) : INT <<AUTO_INCREMENT>>
    --
    NOT_NULL(name) : VARCHAR(100) <<UNIQUE>>
    description : VARCHAR(255)
}

TABLE(visits) {
    PRIMARY_KEY(id) : INT <<AUTO_INCREMENT>>
    --
    FOREIGN_KEY(person_id) : INT <<FK, NOT NULL>>
    NOT_NULL(visit_date) : DATE
    NOT_NULL(visit_time) : TIME
    NOT_NULL(duration) : INT
    NOT_NULL(theme) : VARCHAR(255)
    content : VARCHAR(1000)
    NOT_NULL(visit_type) : ENUM
    NOT_NULL(rfid_chip) : VARCHAR(50) <<UNIQUE>>
}

note right of visits::visit_type
  ENUM Werte:
  'Privat', 'Offiziell'
end note

TABLE(movements) {
    PRIMARY_KEY(id) : INT <<AUTO_INCREMENT>>
    --
    FOREIGN_KEY(visit_id) : INT <<FK, NOT NULL>>
    FOREIGN_KEY(location_id) : INT <<FK, NOT NULL>>
    NOT_NULL(entry_time) : DATETIME
    exit_time : DATETIME
}
